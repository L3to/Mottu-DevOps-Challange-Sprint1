## Para rodar no azure CLI
touch abrePorta.sh
nano abrePorta.sh
chmod 744 abrePorta.sh
./abrePorta.sh

#!/bin/bash

grupox="rg-sll"
vmnamex="sll-vm"

porta="8080"
prioridade=100

az vm open-port --resource-group "$grupox" --name "$vmnamex" --port "$porta" --priority "$prioridade"

echo "A porta $porta foi aberta com sucesso."
